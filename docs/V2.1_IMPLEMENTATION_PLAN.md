### Test Framework Structure

```rust
// core/src/testing/connector_test.rs

pub struct ConnectorTest {
    name: String,
    connector_id: String,
    test_type: TestType,
    expected_capability: Capability,
    test_data: Option<Vec<Record>>,
    assertions: Vec<Assertion>,
}

pub enum TestType {
    Connection,           // Can we connect?
    SchemaDetection,      // Can we detect schema?
    Read,                 // Can we read data?
    Write,                // Can we write data?
    BatchWrite,           // Can we batch write?

}

pub struct TestResult {
    test_name: String,
    passed: bool,
    duration_ms: u64,
    error: Option<String>,
    metrics: ConnectorMetrics,
}
```

### Test Harness Implementation

```rust
// core/src/testing/harness.rs

pub struct ConnectorTestHarness {
    registry: ConnectorRegistry,
    test_database: TestDatabase,
    metrics_collector: MetricsCollector,
}

impl ConnectorTestHarness {
    pub async fn test_connector(&self, connector_id: &str) -> ConnectorTestReport {
        let tests = self.load_tests(connector_id);
        let mut results = Vec::new();
        
        for test in tests {
            let result = match test.test_type {
                TestType::Connection => self.test_connection(&test).await,
                TestType::SchemaDetection => self.test_schema(&test).await,
                TestType::Read => self.test_read(&test).await,
                TestType::Write => self.test_write(&test).await,
                TestType::RateLimit => self.test_rate_limit(&test).await,
                TestType::ErrorRecovery => self.test_error_recovery(&test).await,
                TestType::Integration => self.test_integration(&test).await,
            };
            results.push(result);
        }
        
        // Generate comprehensive report
        ConnectorTestReport {
            connector_id: connector_id.to_string(),
            total_tests: results.len(),
            passed: results.iter().filter(|r| r.passed).count(),
            failed: results.iter().filter(|r| !r.passed).count(),
            results,
            timestamp: Utc::now(),
        }
    }
    
    // Individual test methods
    async fn test_connection(&self, test: &ConnectorTest) -> TestResult { ... }
    async fn test_schema(&self, test: &ConnectorTest) -> TestResult { ... }
    async fn test_read(&self, test: &ConnectorTest) -> TestResult { ... }
    async fn test_write(&self, test: &ConnectorTest) -> TestResult { ... }
    async fn test_rate_limit(&self, test: &ConnectorTest) -> TestResult { ... }
    async fn test_error_recovery(&self, test: &ConnectorTest) -> TestResult { ... }
    async fn test_integration(&self, test: &ConnectorTest) -> TestResult { ... }
}

pub struct ConnectorTestReport {
    pub connector_id: String,
    pub total_tests: usize,
    pub passed: usize,
    pub failed: usize,
    pub results: Vec<TestResult>,
    pub timestamp: DateTime<Utc>,
}
```

## Connection Management Enhancements

### Connection Pooling Optimizations

```rust
// core/src/connections/pool.rs

pub struct ConnectionPool {
    connections: HashMap<String, VecDeque<Connection>>,
    max_connections: usize,
    idle_timeout: Duration,
    health_check_interval: Duration,
}

impl ConnectionPool {
    pub async fn get(&mut self, connector_id: &str) -> Connection {
        if let Some(conn) = self.connections.get_mut(connector_id) {
            if let Some(pooled) = conn.pop_front() {
                if self.is_healthy(&pooled) {
                    return pooled;
                }
            }
        }
        // Create new connection
        self.create_connection(connector_id).await
    }
    
    pub async fn release(&mut self, connector_id: &str, conn: Connection) {
        if let Some(pool) = self.connections.get_mut(connector_id) {
            if pool.len() < self.max_connections {
                pool.push_back(conn);
            }
        }
    }
    
    fn is_healthy(&self, conn: &Connection) -> bool {
        // Check idle timeout
        conn.last_used.elapsed() < self.idle_timeout
        // Could add health check ping
    }
}
```

### Credential Encryption

```rust
// core/src/connections/credentials.rs

pub struct EncryptedCredentials {
    algorithm: String,  // AES-256-GCM
    encrypted: Vec<u8>,
    nonce: Vec<u8>,
    key_id: String,  // Reference to key management system
}

impl EncryptedCredentials {
    pub fn encrypt(credentials: &Credentials, key_manager: &KeyManager) -> Self {
        let key = key_manager.get_key("default");
        let cipher = Aes256Gcm::new(key);
        let nonce = Nonce::from_slice(&[0u8; 12]);
        
        let encrypted = cipher.encrypt(nonce, credentials.as_bytes().as_ref())
            .expect("encryption failed");
        
        EncryptedCredentials {
            algorithm: "AES-256-GCM".to_string(),
            encrypted,
            nonce: nonce.to_vec(),
            key_id: "default".to_string(),
        }
    }
    
    pub fn decrypt(&self, key_manager: &KeyManager) -> Result<Credentials, CryptoError> {
        let key = key_manager.get_key(&self.key_id);
        let cipher = Aes256Gcm::new(key);
        let nonce = Nonce::from_slice(&self.nonce);
        
        let decrypted = cipher.decrypt(nonce, self.encrypted.as_ref())?;
        Ok(Credentials::from_bytes(&decrypted))
    }
}
```

```rust
pub struct ConnectionHealthMonitor {

impl ConnectionHealthMonitor {
    pub async fn monitor(&self, pool: &ConnectionPool) {
        loop {
            for connector_id in pool.connector_ids() {
                let healthy = self.check_health(connector_id).await;
                pool.mark_health(connector_id, healthy);
            }
            tokio::time::sleep(self.interval).await;
```
